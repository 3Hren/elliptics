CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )
PROJECT( ELLIPTICS_DOC )

SET( ELLIPTICS_DOC_LANG English CACHE STRING "Documentation language")
SET( ELLIPTICS_DOC_BRIEF_MEMBER_DESC NO CACHE STRING "Include brief member descriptions after the members that are listed in the file and class documentation" )
SET( ELLIPTICS_DOC_EXTRACT_PRIVATE NO CACHE STRING "" )
SET( ELLIPTICS_DOC_EXTRACT_STATIC NO CACHE STRING "" )
SET( ELLIPTICS_DOC_QUIET NO CACHE STRING "Quiet documentation generation" )
SET( ELLIPTICS_DOC_FILE_PATTERNS "*.h *.hpp *.qdoc" CACHE STRING "Space separated wildcard patterns" )
SET( ELLIPTICS_DOC_EXCLUDE_PATTERNS "*_p.h *_p.hpp" CACHE STRING "Space separated wildcard patterns" )
SET( ELLIPTICS_DOC_EXCLUDE_SYMBOLS d_ptr CACHE STRING "Space separated wildcard patterns" )
SET( ELLIPTICS_DOC_SOURCE_BROWSER NO CACHE STRING "" )
SET( ELLIPTICS_DOC_INLINE_SOURCES NO CACHE STRING "" )
SET( ELLIPTICS_DOC_REFERENCED_BY_RELATION NO CACHE STRING "For each documented function all documented functions referencing it will be listed" )
SET( ELLIPTICS_DOC_REFERENCES_RELATION NO CACHE STRING "For each documented function all documented entities called/used by that function will be listed" )
SET( ELLIPTICS_DOC_GENERATE_ECLIPSEHELP NO CACHE STRING "" )
SET( ELLIPTICS_DOC_GENERATE_TREEVIEW YES CACHE STRING "" )
SET( ELLIPTICS_DOC_TREEVIEW_WIDTH 250 CACHE STRING "Set width if GENERATE_TREEVIEW is YES" )
SET( ELLIPTICS_DOC_EXPAND_ONLY_PREDEF NO CACHE STRING "" )
SET( ELLIPTICS_DOC_PREDEFINED Q_QDOC CACHE STRING "Predefined preprocessor macros names (like gcc -D ...)" )
SET( ELLIPTICS_DOC_CLASS_DIAGRAMS YES CACHE STRING "" )
SET( ELLIPTICS_DOC_HIDE_UNDOC_RELATIONS YES CACHE STRING "" )
SET( ELLIPTICS_DOC_HAVE_DOT NO CACHE STRING "Advanced graph visualization" )
SET( ELLIPTICS_DOC_UML_LOOK NO CACHE STRING "" )
SET( ELLIPTICS_DOC_CALL_GRAPH NO CACHE STRING "Works only if HAVE_DOT is YES. Enable only if you know what you are doing!" )
SET( ELLIPTICS_DOC_CALLER_GRAPH NO CACHE STRING "Works only if HAVE_DOT is YES. Enable only if you know what you are doing!" )
SET( ELLIPTICS_DOC_DOT_GRAPH_MAX_NODES 50 CACHE STRING "" )
SET( ELLIPTICS_DOC_MAX_DOT_GRAPH_DEPTH 0 CACHE STRING "0 is equal to no restriction" )

IF ( ELLIPTICS_DOC_SOURCE_BROWSER STREQUAL YES )
   SET (ELLIPTICS_DOC_LAYOUT_TAB_GUARD_GL_OP "")
   SET (ELLIPTICS_DOC_LAYOUT_TAB_GUARD_GL_ED "")
ELSE ()
   SET (ELLIPTICS_DOC_LAYOUT_TAB_GUARD_GL_OP "<!--")
   SET (ELLIPTICS_DOC_LAYOUT_TAB_GUARD_GL_ED "-->")
ENDIF ()

MARK_AS_ADVANCED(
   ELLIPTICS_DOC_BRIEF_MEMBER_DESC
   ELLIPTICS_DOC_EXTRACT_PRIVATE
   ELLIPTICS_DOC_EXTRACT_STATIC
   ELLIPTICS_DOC_QUIET
   ELLIPTICS_DOC_FILE_PATTERNS
   ELLIPTICS_DOC_EXCLUDE_PATTERNS
   ELLIPTICS_DOC_EXCLUDE_SYMBOLS
   ELLIPTICS_DOC_INLINE_SOURCES
   ELLIPTICS_DOC_REFERENCED_BY_RELATION
   ELLIPTICS_DOC_REFERENCES_RELATION
   ELLIPTICS_DOC_TREEVIEW_WIDTH
   ELLIPTICS_DOC_EXPAND_ONLY_PREDEF
   ELLIPTICS_DOC_PREDEFINED
   ELLIPTICS_DOC_HIDE_UNDOC_RELATIONS
   ELLIPTICS_DOC_CALL_GRAPH
   ELLIPTICS_DOC_CALLER_GRAPH
   ELLIPTICS_DOC_DOT_GRAPH_MAX_NODES
   ELLIPTICS_DOC_MAX_DOT_GRAPH_DEPTH
)
